    <script>
      // 🔄 Realtime Application State Manager
      let globalOrders = [];
      let currentActiveFilter = 'Done'; // Default View: Today's Orders (Done)

      // Backend server configuration
      const socket = io("http://127.0.0.1:5000", {
        transports: ["polling", "websocket"] 
      });

      // Connection lifecycle event
      socket.on("connect", function() {
        console.log("[SOCKET] Internet connection established with Flask server.");
        const loggedUser = localStorage.getItem("loggedUser") || "default_user";
        socket.emit("get_initial_list", { user_id: loggedUser });
      });

      // Real-time Update Listener
      socket.on("update_email_list", function(payload) {
        console.log("Data received from backend:", payload);
          
        if (payload && payload.emails) {
          globalOrders = payload.emails;
          
          // 🚀 Dynamically update Table & Sidebar Queue Views simultaneously
          renderOrdersTable();
          renderSidebarQueue(payload.emails);
        }
      });

      socket.on("new_order", function(order){
        console.log("NEW ORDER ALERT:", order);
      });

      // 🎯 Interactive Filters Renderer
      function filterOrdersByStatus(status) {
        currentActiveFilter = status;
        
        const titleElement = document.getElementById("orders-table-title");
        if (titleElement) {
          if (status === 'Done') {
            titleElement.innerText = "Today's Confirmed Orders";
          } else if (status === 'Running') {
            titleElement.innerText = "Pending & Active Processing Orders";
          } else if (status === 'Failed') {
            titleElement.innerText = "Cancelled / Failed Orders";
          }
        }
        renderOrdersTable();
      }

      // 🛠️ Render Dynamic Realtime Orders Table
      function renderOrdersTable() {
        const tbody = document.getElementById("realtime-orders-table-body");
        if (!tbody) return;

        // Filter global orders based on selection
        const filtered = globalOrders.filter(order => {
          if (currentActiveFilter === 'Running') {
            return order.status === 'Running' || order.status === 'Pending';
          }
          return order.status === currentActiveFilter;
        });

        if (filtered.length === 0) {
          let statusFriendly = currentActiveFilter === 'Done' ? 'Confirmed' : (currentActiveFilter === 'Running' ? 'Pending' : 'Cancelled');
          tbody.innerHTML = `
            <tr>
              <td colspan="4" class="text-center text-muted py-5">
                <i class="mdi mdi-information-outline me-2" style="font-size: 18px; vertical-align: middle;"></i>
                No active ${statusFriendly.toLowerCase()} orders found.
              </td>
            </tr>
          `;
          return;
        }

        tbody.innerHTML = ""; // Clear current rows

        filtered.forEach(order => {
          let badgeClass = "badge-opacity-warning";
          let statusLabel = order.status;

          if (order.status === "Done") {
            badgeClass = "badge-opacity-success";
            statusLabel = "Confirmed";
          } else if (order.status === "Running") {
            badgeClass = "badge-opacity-primary";
            statusLabel = "Processing";
          } else if (order.status === "Failed") {
            badgeClass = "badge-opacity-danger";
            statusLabel = "Cancelled";
          }

          const tr = document.createElement("tr");
          tr.style.borderBottom = "1px solid #f3f3f3";

      // 🛠️ Sidebar List Renderer
      function renderSidebarQueue(emails) {
        const queueContainer = document.getElementById("email-queue-list");
        if (!queueContainer) return;

        if (!emails || emails.length === 0) {
          queueContainer.innerHTML = '<li class="text-muted p-3 text-center">No active emails loaded yet...</li>';
          return;
        }

        queueContainer.innerHTML = ""; // Clear existing view

        emails.forEach((item) => {
          let badgeClass = "badge-opacity-warning"; // Default: Pending
          if (item.status === "Running") badgeClass = "badge-opacity-primary";
          if (item.status === "Done") badgeClass = "badge-opacity-success";
          if (item.status === "Failed") badgeClass = "badge-opacity-danger";

          const li = document.createElement("li");
          li.className = "d-block";
          li.style.borderBottom = "1px solid #f3f3f3";
          li.style.padding = "12px 0";
          
          li.innerHTML = `
            <div class="w-100 d-flex align-items-center justify-content-between mb-0" style="padding: 2px 0;">
              <label class="mb-0" style="max-width: 60%; overflow: hidden; text-overflow: ellipsis; white-space: nowrap; cursor: default;">
                <input type="checkbox" ${item.status === 'Done' ? 'checked' : ''} disabled style="transform: scale(1.1); vertical-align: middle; cursor: not-allowed;"> 
                <strong style="color: #2b2b2b; font-size: 13.5px; margin-left: 8px; vertical-align: middle;">${item.email}</strong>
                <br>
                <span style="font-size: 11px; color: #888; margin-left: 24px; display: inline-block; margin-top: 2px;">Firebase ID: ${item.id}</span>
              </label>
              
              <div class="d-flex align-items-center" style="gap: 10px;">
                <span class="badge ${badgeClass}" style="font-size:11px; padding: 5px 10px;">${item.status}</span>
                <button class="btn btn-sm btn-primary text-white" 
                        style="padding: 6px 12px; font-size: 12px; font-weight: bold; border-radius: 4px;"
                        onclick="triggerDirectAutomation('${item.email}')"
                        ${item.status === 'Running' ? 'disabled' : ''}>
                  ${item.status === 'Running' ? 'Running...' : 'Sign In 🚀'}
                </button>
              </div>
            </div>
          `;
          queueContainer.appendChild(li);
        });
      }

      // Execute Workflow Endpoint Trigger
      function triggerDirectAutomation(targetEmail) {
        console.log("Triggering automation workflow for:", targetEmail);
        
        fetch("http://127.0.0.1:5000/run-automation-direct", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ email: targetEmail })
        })
        .then(res => res.json())
        .then(data => console.log("Backend automation trigger acknowledgement:", data))
        .catch(err => console.error("Pipeline trigger failure:", err));
      }
